const passport = require('passport'),
    /**
     * Passport configuration for authentication strategies.
     *
     * - LocalStrategy: Authenticates users using username and password.
     * - JWTStrategy: Authenticates users using JWT tokens.
     *
     * @module passport
     */
    LocalStrategy = require('passport-local').Strategy,
    Models = require('./models.js'),
    passportJWT = require('passport-jwt');

require('dotenv').config();

let Users = Models.User,
    JWTStrategy = passportJWT.Strategy,
    ExtractJWT = passportJWT.ExtractJwt;


/**
 * Local authentication strategy for username and password login.
 *
 * @name LocalStrategy
 * @function
 * @memberof module:passport
 * @param {string} username - Username provided by the user
 * @param {string} password - Password provided by the user
 * @param {Function} callback - Callback to pass control to next middleware
 * @returns {Object|boolean} User object if authentication is successful, false otherwise
 */
passport.use(
    new LocalStrategy(
        {
            usernameField: 'username',
            passwordField: 'password',
        },
        async (username, password, callback) => {
            await Users.findOne({ username: username })
                .then((user) => {
                    if (!user) {
                        return callback(null, false, {
                            message: 'Incorrect username or password.',
                        });
                    }
                    if (!user.validatePassword(password)) {
                        return callback(null, false, { message: 'Incorrect password.' });
                    }
                    return callback(null, user);
                })
                .catch((error) => {
                    if (error) {
                        return callback(error);
                    }
                })
        }
    )
);

/**
 * JWT authentication strategy for verifying JWT tokens.
 *
 * @name JWTStrategy
 * @function
 * @memberof module:passport
 * @param {Object} jwtPayload - Decoded JWT payload
 * @param {Function} callback - Callback to pass control to next middleware
 * @returns {Object|null} User object if token is valid, null otherwise
 */
passport.use(new JWTStrategy({
    jwtFromRequest: ExtractJWT.fromAuthHeaderAsBearerToken(),
    secretOrKey: process.env.JWT_SECRET
},
    async (jwtPayload, callback) => {
        return await Users.findById(jwtPayload._id)
            .then((user) => {
                return callback(null, user);
            })
            .catch((error) => {
                return callback(error)
            });
    }
));
